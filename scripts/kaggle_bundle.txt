with open('engine/orchestrator.py', 'w') as f:
    f.write('import os\nimport subprocess\nimport json\nimport zipfile\nimport traceback\nimport numpy as np\nimport shutil\nfrom pathlib import Path\nfrom engine.config import logger, GPU_0, GPU_1, OUTPUT_DIR\nfrom engine.model_manager import clear_vram, get_beat_models\nfrom engine.processing import get_chords_btc_max_accuracy\n\n# orchestrate dual-gpu pipeline\ndef remix_audio_dual_gpu(audio_path, engine_choice, stems_mode):\n    try:\n        if not audio_path: \n            return [None]*11\n            \n        # clean output dir\n        if os.path.exists(OUTPUT_DIR):\n            logger.info("[CLEANUP] Pre-clearing separated directory...")\n            shutil.rmtree(OUTPUT_DIR, ignore_errors=True)\n        os.makedirs(OUTPUT_DIR, exist_ok=True)\n\n        clear_vram()\n        \n        is_roformer = "RoFormer" in engine_choice\n        stem_dir = Path(OUTPUT_DIR) / "stems"\n        stem_dir.mkdir(parents=True, exist_ok=True)\n        \n        if is_roformer:\n            # run roformer separation\n            logger.info(f"Starting BS-RoFormer (6 Stems) Separation on {GPU_0}...")\n            # specific 6-stem BS-roformer SW model\n            model_name = "BS-Roformer-SW.ckpt"\n            subprocess.run([\n                "audio-separator", audio_path,\n                "--model_filename", model_name,\n                "--output_dir", str(stem_dir),\n                "--output_format", "WAV",\n                "--mdxc_segment_size", "256",    \n                "--mdxc_overlap", "4"\n            ], check=True, env={**os.environ, "CUDA_VISIBLE_DEVICES": GPU_0[-1]})\n            \n            v, d, b, o, g, p = None, None, None, None, None, None\n            \n            # map output files\n            for file in stem_dir.glob("*.wav"):\n                fname = file.name.lower()\n                if "(vocals)" in fname or "_vocals_" in fname: v = str(file)\n                elif "(drums)" in fname or "_drums_" in fname: d = str(file)\n                elif "(bass)" in fname or "_bass_" in fname: b = str(file)\n                elif "(other)" in fname or "_other_" in fname: o = str(file)\n                elif "(guitar)" in fname or "_guitar_" in fname: g = str(file)\n                elif "(piano)" in fname or "_piano_" in fname: p = str(file)\n                \n            # fallback to exact names\n            if not v: v = str(stem_dir/"vocals.wav") if (stem_dir/"vocals.wav").exists() else v\n            if not d: d = str(stem_dir/"drums.wav") if (stem_dir/"drums.wav").exists() else d\n            if not b: b = str(stem_dir/"bass.wav") if (stem_dir/"bass.wav").exists() else b\n            if not o: o = str(stem_dir/"other.wav") if (stem_dir/"other.wav").exists() else o\n            \n            # final existence check\n            if not v or not os.path.exists(v):\n                logger.warning(f"[ROFORMER] Primary stem discovery failed. Listing directory...")\n                for f in stem_dir.glob("*"): logger.info(f"Found file: {f.name}")\n            \n            if stems_mode == "4 Stems":\n                g, p = None, None\n        else:\n            # run demucs separation\n            model_name = "htdemucs_ft" if stems_mode == "4 Stems" else "htdemucs_6s"\n            logger.info(f"Starting Demucs Separation on {GPU_0}...")\n            subprocess.run(["demucs", "-d", GPU_0, "-n", model_name, audio_path, "-o", OUTPUT_DIR], check=True)\n        \n            demucs_stem_dir = Path(OUTPUT_DIR) / model_name / Path(audio_path).stem\n            for f in demucs_stem_dir.glob("*.wav"):\n                shutil.move(str(f), str(stem_dir / f.name))\n                \n            v, d, b, o = [str(stem_dir/f"{s}.wav") for s in ["vocals", "drums", "bass", "other"]]\n            g = str(stem_dir/"guitar.wav") if stems_mode == "6 Stems" and (stem_dir/"guitar.wav").exists() else None\n            p = str(stem_dir/"piano.wav") if stems_mode == "6 Stems" and (stem_dir/"piano.wav").exists() else None\n        \n        # track song beats\n        logger.info("Starting Madmom Beat Tracking on CPU...")\n        BEAT_FEAT, BEAT_DECODE = get_beat_models()\n        beat_activations = BEAT_FEAT(audio_path)\n        beats = BEAT_DECODE(beat_activations).tolist()\n        tempo = round(60 / np.median(np.diff(beats))) if len(beats) > 1 else 120\n        \n        # recognize song chords\n        logger.info(f"Starting MAX ACCURACY BTC Chord Recognition on {GPU_1}...")\n        chord_data, reasoning = get_chords_btc_max_accuracy(audio_path, beats, tempo=tempo, bass_audio_path=b, other_audio_path=o)\n        \n        sheet_text = f"MAX ACCURACY DUAL-T4 REPORT (VITERBI MODE)\\nBPM: {tempo}\\n" + "="*30 + "\\n\\n"\n        for c in chord_data: sheet_text += f"[{c[\'time\']}s] {c[\'chord\']}\\n"\n        \n        # package results zip\n        zip_p = "/kaggle/working/Kaggle_Dual_T4_Max_Accuracy_Results.zip"\n        with zipfile.ZipFile(zip_p, \'w\') as z:\n            chords_file = stem_dir/"chords.json"\n            with open(chords_file, "w") as f: json.dump(chord_data, f, indent=2)\n            z.write(chords_file, arcname="chords.json")\n            reasoning_file = stem_dir/"reasoning.txt"\n            with open(reasoning_file, "w") as f: f.write(reasoning)\n            z.write(reasoning_file, arcname="reasoning.txt")\n        \n        clear_vram()\n        return v, d, b, o, g, p, chord_data, {"beats": beats, "tempo": tempo}, sheet_text, zip_p, reasoning\n\n    except Exception as e:\n        logger.error(f"Error in remix_audio_dual_gpu: {e}")\n        err_msg = traceback.format_exc()\n        return [None]*6 + [[], {}, f"FATAL ERROR:\\n{str(e)}\\n\\n{err_msg}", None, f"Pipeline Crashed:\\n{str(e)}"]\n')
